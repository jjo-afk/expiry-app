const today = new Date(); today.setHours(0,0,0,0);
function daysLeft(d){if(!d)return null;const x=new Date(d);x.setHours(0,0,0,0);return Math.ceil((x-today)/86400000);}
function statusColor(d){if(d===null)return"#aaa";if(d<0)return"#c62828";if(d<=7)return"#e65100";if(d<=30)return"#f9a825";return"#2e7d32";}
function statusLabel(d){if(d===null)return"미입력";if(d<0)return`${Math.abs(d)}일 초과`;if(d===0)return"오늘";return`D-${d}`;}
function paoInfo(openedDate,sub){
  if(!openedDate)return null;
  const opened=new Date(openedDate);opened.setHours(0,0,0,0);
  const exp=new Date(opened);exp.setMonth(exp.getMonth()+(PAO[sub]||12));
  const pct=Math.min(100,Math.max(0,Math.round((today-opened)/(exp-opened)*100)));
  return{pct,rem:Math.ceil((exp-today)/86400000),months:PAO[sub]||12};
}

function PaoBar({openedDate,sub}){
  const p=paoInfo(openedDate,sub);if(!p)return null;
  const c=p.pct>=100?"#c62828":p.pct>=75?"#e65100":p.pct>=50?"#f9a825":"#2e7d32";
  return(
    <div style={{marginTop:8}}>
      <div style={{display:"flex",justifyContent:"space-between",fontSize:11,color:"#888",marginBottom:3}}>
        <span>개봉 후 사용기한 ({p.months}개월)</span>
        <span style={{color:c,fontWeight:600}}>{p.pct>=100?"기한 초과":`${p.pct}% · ${p.rem}일 남음`}</span>
      </div>
      <div style={{background:"#eee",borderRadius:6,height:8,overflow:"hidden"}}>
        <div style={{width:`${p.pct}%`,height:"100%",background:c,borderRadius:6}}/>
      </div>
    </div>
  );
}

            <span style={{position:"absolute",left:12,top:"50%",transform:"translateY(-50%)",color:"#bbb",fontSize:15}}>🔍</span>
            <input style={si} placeholder="제품명, 분류, 메모 검색" value={searchQ} onChange={e=>setSearchQ(e.target.value)}/>
          </div>
          <div style={{display:"flex",gap:7,marginBottom:14,overflowX:"auto",paddingBottom:2}}>
            {ftabBtn("all",`전체 (${items.length})`)}
            {ftabBtn("food","🍽 식품")}
            {ftabBtn("bathroom","🛁 욕실")}
            {ftabBtn("cosmetics","💄 화장품")}
          </div>
          {listed.length===0
            ?<div style={{textAlign:"center",color:"#bbb",padding:"48px 0",fontSize:15}}>{q?"검색 결과가 없습니다":"등록된 제품이 없습니다"}</div>
            :listed.map(item=><ItemCard key={item.id} item={item} onSave={form=>handleSave(form,item.id)} onDelete={handleDelete}/>)
          }
          <button onClick={()=>setModal(true)} style={{width:"100%",padding:13,borderRadius:12,fontSize:15,fontWeight:700,border:"none",cursor:"pointer",background:"#222",color:"#fff",marginTop:4}}>+ 새 제품 등록</button>
        </>}

        {/* 메모 */}
        {tab==="notes" && <>
          <div style={{position:"relative",marginBottom:14}}>
            <span style={{position:"absolute",left:12,top:"50%",transform:"translateY(-50%)",color:"#bbb",fontSize:15}}>🔍</span>
            <input style={si} placeholder="메모 내용 검색" value={searchQ} onChange={e=>setSearchQ(e.target.value)}/>
          </div>
          {noted.length===0
            ?<div style={{textAlign:"center",color:"#bbb",padding:"48px 0",fontSize:15}}>메모가 있는 제품이 없습니다<br/><span style={{fontSize:13}}>제품 등록 시 특이사항을 메모해보세요</span></div>
            :noted.map(item=>{
              const cfg=CATS[item.category];const d=daysLeft(item.expiry);
              return(
                <div key={item.id} style={{background:"#fff",border:"1px solid #eee",borderLeft:"3px solid #f59e0b",borderRadius:14,padding:"14px 16px",marginBottom:12}}>
                  <div style={{display:"flex",justifyContent:"space-between",alignItems:"flex-start"}}>
                    <div>
                      <div style={{fontWeight:700,fontSize:15}}>{item.name}</div>
                      <div style={{fontSize:11,color:"#aaa",margin:"3px 0 8px"}}>{cfg.icon} {cfg.label} · {item.sub}</div>
                    </div>
                    <span style={{fontWeight:700,fontSize:12,color:statusColor(d)}}>{statusLabel(d)}</span>
                  </div>
                  <div style={{fontSize:13,color:"#555",lineHeight:1.65,whiteSpace:"pre-wrap"}}>{item.note}</div>
                </div>
              );
            })
          }
        </>}
      </div>

      <Modal open={modal} onClose={()=>setModal(false)} onSave={form=>handleSave(form)} initial={BLANK} key={String(modal)}/>
    </div>
  );
}
